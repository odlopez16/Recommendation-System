# Ascendente (A-Z)
GET /api/products?sort_by=name&order=asc

# Descendente (Z-A)
GET /api/products?sort_by=name&order=desc

# Precio más bajo primero
GET /api/products?sort_by=price&order=asc

# Precio más alto primero
GET /api/products?sort_by=price&order=desc

# Buscar productos que contengan "laptop"
GET /api/products?search=laptop

# Buscar laptops, ordenadas por precio descendente, 10 productos por página
GET /api/products?search=laptop&sort_by=price&order=desc&limit=10

# Productos de la categoría "electronics", ordenados por nombre
GET /api/products?category=electronics&sort_by=name&order=asc
